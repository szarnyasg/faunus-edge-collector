# basic install
wget http://s3.thinkaurelius.com/downloads/titan/titan-all-0.3.2.zip
wget http://s3.thinkaurelius.com/downloads/faunus/faunus-0.3.2.zip
unzip titan-all-0.3.2.zip
unzip faunus-0.3.2.zip
sudo apt-get install -y software-properties-common

sudo add-apt-repository -y ppa:webupd8team/java
sudo apt-get update
sudo apt-get install -y oracle-java7-installer



# setup IQD stuff
mkdir mr
cd mr
wget https://www.dropbox.com/s/socpwoczg06mfea/mr.zip
wget https://www.dropbox.com/s/e5dly34vdhzuhaz/testBig_User_16.faunus-graphson.zip
unzip mr.zip
unzip testBig_User_16.faunus-graphson.zip 
export FAUNUS_HOME=~/faunus-0.3.2



# run Titan
cd ~/titan-all-0.3.2; export JAVA_OPTIONS="-Xms4G -Xmx4G -javaagent:/home/szarnyasg/titan-all-0.3.2/lib/jamm-0.2.5.jar"; rm -rf /tmp/cassandra/; bin/titan.sh config/titan-server-rexster.xml config/titan-server-cassandra.properties



# run IQD stuff
cd ~/mr
time ~/faunus-0.3.2/bin/gremlin.sh < faunus-edgetypes.txt



# setup Cassandra
wget http://xenia.sote.hu/ftp/mirrors/www.apache.org/cassandra/1.2.9/apache-cassandra-1.2.9-bin.tar.gz
tar xf apache-cassandra-1.2.9-bin.tar.gz 
sudo mkdir /var/log/cassandra
sudo mkdir /var/lib/cassandra
sudo chown szarnyasg:vcl /var/log/cassandra
sudo chown szarnyasg:vcl /var/lib/cassandra



# run Cassandra
cd ~/apache-cassandra-1.2.9/
bin/cassandra -f
